117  res[0] = -alpha * G * A; // dG/dt=-alpha*G*A

120  res[3] = ((1 - kappa) / qq) * A + ((1 - theta) / pp) * K; // dR/dt= (1-kappa)/q *A +(1 -theta)/pp *K
121  res[4] = (theta / pp) * K; // dV/dt = theta/pp * K
